\section{Shadings Library}
\label{section-library-shadings}

\begin{pgflibrary}{shadings}
  The package defines a number of shadings in addition to the ball and
  axis shadings that are available by default.
\end{pgflibrary}

To be written...
\begin{codeexample}[]
\begin{tikzpicture}
  \shade[shading=color wheel white center] (0,0) circle (1);
  \shade[shading=color wheel black center] (0,-2) circle (1);
  \shade[shading=color wheel,even odd rule](0,-4) circle (1)
                                           (0,-4) circle (.5);
  \shade[upper left=red,upper right=green,
         lower left=blue,lower right=yellow]
                                      (-1,-6) rectangle ++(2,-2);
  \shade[shading=Mandelbrot set]      (-1,-8) rectangle ++(2,-2);
\end{tikzpicture}
\end{codeexample}
