\usetikzlibrary{circuits.ee}
\usepgflibrary{shapes.gates.ee.IEC}


%
% When this lib is loaded, all symbols default to IEC
%

\tikzset{
  circuit ee IEC/.style=
  {
    circuit,
    resistor graphic    = resistor IEC graphic,
    inductor graphic    = inductor IEC graphic,
    capacitor graphic   = capacitor IEC graphic,
    link graphic        = link IEC graphic,
    connection graphic  = connection IEC graphic,
    ground graphic      = ground IEC graphic,
    voltage graphic     = voltage IEC graphic,
    diode graphic       = diode IEC graphic,
    Zener diode graphic = Zener diode IEC graphic,
    bulb graphic        = bulb IEC graphic,
  }
}



%
% Resistors
%

\tikzset{
  resistor IEC graphic/.style={
    circuit symbol open,
    circuit symbol size=width 5 height 1.25,
    shape=rectangle,
    transform shape
  },
  var resistor IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width 4.8 height 0.8,
    shape=var resistor IEC,
    transform shape,
    outer sep=0pt,
    cap=round
  }
}



%
% Inductors
%

\tikzset{
  inductor IEC graphic/.style={
    circuit symbol filled,
    circuit symbol size=width 5 height 1.25,
    transform shape,
    shape=rectangle
  },
  var inductor IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width 5 height .5,
    shape=var inductor IEC,
    transform shape,
    outer sep=0pt,
    cap=round
  }
}



%
% Capacitors
%

\tikzset{
  capacitor IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width .5 height 2,
    transform shape,
    shape=capacitor IEC
  }
}



%
% Ground
%

\tikzset{
  ground IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width .75 height 2,
    transform shape,
    shape=ground IEC
  }
}




%
% Voltage
%

\tikzset{
  voltage IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width .5 height 2.5,
    transform shape,
    shape=voltage IEC
  }
}



%
% Diodes
%

\tikzset{
  diode IEC graphic/.style={
    circuit symbol open,
    circuit symbol size=width 1.25 height 1.25,
    transform shape,
    shape=diode IEC
  },
  Zener diode IEC graphic/.style={
    diode IEC graphic,
    shape=Zener diode IEC
  }
}



%
% Links and connections
%

\tikzset{
  link IEC graphic/.style={
    circuit symbol filled,
    circuit symbol size=width .5 height .5,
    shape=circle,
    outer sep=.25\pgflinewidth
  },
  connection IEC graphic/.style={
    circuit symbol open,
    circuit symbol size=width 1 height 1,
    shape=circle,
    outer sep=.25\pgflinewidth
  }
}


%
% Bulbs
%

\tikzset{
  bulb IEC graphic/.style={
    circuit symbol lines,
    circuit symbol size=width 1.5 height 1.5,
    shape=bulb IEC,
    outer sep=.25\pgflinewidth
  }
}



\endinput
