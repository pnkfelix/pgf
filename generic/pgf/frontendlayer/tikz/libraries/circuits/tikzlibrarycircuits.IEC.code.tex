\usetikzlibrary{circuits}


%
% When this lib is loaded, all symbols default to IEC
%

\tikzset{
  circuit IEC/.style=
  {
    circuit,
    resistor/.style =         {resistor IEC={##1}},
    inductor/.style =         {inductor IEC={##1}},
    link/.style =             {link IEC={##1}},
    connection/.style =       {connection IEC={##1}},
  }
}



%
% Resistors
%

\tikzset{
  resistor node IEC/.style={
    every circuit symbol/.try,
    open symbol,
    minimum width=40pt,minimum height=15pt,inner sep=0pt,
    every resistor node/.try
  },
  resistor IEC/.style={handle circuit symbol={transform shape,resistor node IEC,#1}}
}



%
% Inductors
%

\tikzset{
  inductor node IEC/.style={
    every circuit symbol/.try,
    filled symbol,
    minimum width=40pt,minimum height=15pt,inner sep=0pt,
    every resistor node/.try
  },
  inductor IEC/.style={handle circuit symbol={transform shape,inductor node IEC,#1}}
}



%
% Links and connections
%

\tikzset{
  link node IEC/.style={
    every circuit symbol/.try,
    filled symbol,
    minimum size=4pt,inner sep=0pt,outer sep=0pt,circle,
    every join node/.try
  },
  link IEC/.style={handle circuit symbol={link node IEC,#1}}
}

\tikzset{
  connection node IEC/.style={
    every circuit symbol/.try,
    open symbol,
    minimum size=6pt,inner sep=0pt,outer sep=0pt,circle,
    every join node/.try
  },
  connection IEC/.style={handle circuit symbol={connection node IEC,#1}}
}




\endinput
