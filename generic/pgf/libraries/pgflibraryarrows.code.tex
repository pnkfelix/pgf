% A square bracket shaped arrow

\pgfarrowsdeclare{[}{]}
{
  \@tempdima=1pt%
  \advance\@tempdima by1.25\pgflinewidth%
  \pgfarrowsleftextend{+-\@tempdima}
  \pgfarrowsrightextend{+.5\pgflinewidth}
}
{
  \@tempdima=2pt%
  \advance\@tempdima by1.5\pgflinewidth%
  \@tempdimb=\@tempdima%
  \advance\@tempdimb by\pgflinewidth%
  \pgfsetdash{}{+0pt}
  \pgfsetmiterjoin
  \pgfsetbuttcap
  \pgfpathmoveto{\pgfqpoint{-.5\@tempdimb}{-1\@tempdima}}
  \pgfpathlineto{\pgfqpoint{0pt}{-1\@tempdima}}
  \pgfpathlineto{\pgfqpoint{0pt}{\@tempdima}}
  \pgfpathlineto{\pgfqpoint{-.5\@tempdimb}{\@tempdima}}
  \pgfusepathqstroke
}

\pgfarrowsdeclarereversed{]}{[}{[}{]}
