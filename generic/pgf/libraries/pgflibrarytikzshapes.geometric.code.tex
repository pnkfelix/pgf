\tikzoption{kite upper vertex angle}{\pgfsetkiteuppervertexangle{#1}}
\tikzoption{kite lower vertex angle}{\pgfsetkitelowervertexangle{#1}}
\tikzoption{kite vertex angles}{%
	\pgfutil@in@,{#1}%
	\ifpgfutil@in@%
		\let\tikz@kitenext\tikz@@processkitevertexangles%
	\else%
		\let\tikz@kitenext\tikz@processkitevertexangles%
	\fi
	\expandafter\tikz@kitenext#1\@@}
\def\tikz@processkitevertexangles#1\@@{\tikz@@processkitevertexangles#1,#1\@@}
\def\tikz@@processkitevertexangles#1,#2\@@{%
	\pgfsetkiteuppervertexangle{#1}%
	\pgfsetkitelowervertexangle{#2}%
}
